``` r
instacart %>%
  group_by(aisle) %>% 
  summarize(n_aisle = n()) %>%
  filter(n_aisle > 10000) %>%
  arrange(desc(n_aisle)) %>%
  ggplot(aes(x = aisle, y = n_aisle, color = aisle)) +
  geom_point() +
  labs(
    title = "Instacart Plot",
    x = "aisle",
    y = "number of orders")
```
